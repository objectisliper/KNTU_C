#include <iostream>
#include <cmath>


int main()
{
	int array[20] = { 1, 2, 3, 4, 5, 0, -5, -4, -3, -2, -1, 0, 3, 0, 10, 9, 0, -11, 21, 0 };
	int  i = 0;
	int sum = 0;

	while (i < 20)
	{
		std::cout << array[i] << "\t";
		i++;
	}
	std::cout << std::endl;
	i = 0;
	int count = 0;
	while (i < 20)
	{
		if (array[i] < 0)
		{
			sum = sum + array[i];
			count++;
		}
		i++;
	};
	double sSum;
	sSum = (double)sqrt(abs(sum / count));
	std::cout << sSum << std::endl;
	for (i = 0; i < 20; i++)
	{
		if (array[i] == 0)
		{
			array[i] =sSum;
		}
	}
	i = 0;
	while (i < 20)
	{
		std::cout << array[i] << "\t";
		i++;
	}
	std::cout << std::endl;

	


	system("pause");
	return 0;
}
