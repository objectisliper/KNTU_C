#include "stdafx.h"
#include <iostream>


int main()
{
	int arr[] = {1, -2, 3, 5, 0, -14, 7, 9, 1, -3, -7, 3, 10, 4, 0};

	//зростання
	std::cout << "Input K: "; //включительно с этого элемента
	int k, l;
	std::cin >> k;
	std::cout << "Input L: "; //невключительно до этого элемента
	std::cin >> l;

	if (k > l)
	{
		std::cout << "Error! Good bye!" << std::endl;
		exit(1);
	}

	for (int i = k; i < l; i++)
	{
		for (int j = l - 1; j > i; j--)
		{
			if (arr[j] < arr[j - 1]) 
			{
				std::swap(arr[j], arr[j - 1]);
			}
		}
	}

	//спадання
	std::cout << "Input M: "; //включительно с этого элемента
	int m, p;
	std::cin >> m;
	std::cout << "Input P: "; //невключительно до этого элемента
	std::cin >> p;

	if (m > p && m < l)
	{
		std::cout << "Error! Good bye!" << std::endl;
		exit(1);
	}

	for (int i = m; i < p; i++)
	{
		for (int j = p - 1; j > i; j--)
		{
			if (arr[j] > arr[j - 1])
			{
				std::swap(arr[j], arr[j - 1]);
			}
		}
	}
	//вывод 
	for (int i = 0; i < 15; i++)
	{
		std::cout << arr[i] << "\t";
	}
	std::cout << std::endl;

	system("pause");
    return 0;
}
