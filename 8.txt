#include <iostream>
#include <cmath>
#include <ctime>


int main()
{
	srand(time(NULL));
	int arr[5][5];
	int sum, count;
	sum = 0;
	count = 0;
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			arr[i][j] = rand() % 50;
		}
	};
	//вывод массива
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			std::cout << arr[i][j] << "\t";
		}
		std::cout << std::endl;
	};

	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			if (j % 2 != 0)
			{
				sum += arr[i][j];
				count++;
			}
		}
	}
	


	std::cout << "Arifmetika = " << sum / count << std::endl;
	system("pause");
	return 0;
}
