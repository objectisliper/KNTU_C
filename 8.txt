#include <stdio.h>
#include <math.h>
#include <time.h>


int main()
{
	
	int arr[5][5];
	int sum, count;
	sum = 0;
	count = 0;
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			arr[i][j] = rand() % 50;
		}
	};
	//вывод массива
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			printf("%d""\t",arr[i][j]);
		}
		
	};

	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 5; j++)
		{
			if (j % 2 != 0)
			{
				sum += arr[i][j];
				count++;
			}
		}
	}
	


	printf("Arifmetika = %d",sum / count);
	
	return 0;
}
